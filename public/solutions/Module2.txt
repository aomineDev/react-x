----------------------------



export default function TarjetaPelicula({
  titulo,
  año,
  generos,
  director,
  calificacion = 5,
  onCambiarCalificacion,
  children
}) {
  return (
    <div style={{ border: '2px solid #ddd', padding: '15px', margin: '10px', borderRadius: '8px' }}>
      <h2>{titulo}</h2>
      <p><strong>Año:</strong> {año}</p>
      <p>
        <strong>Géneros:</strong> {generos.map((genero, index) => (
          <span key={index}>{genero}{index < generos.length - 1 ? ', ' : ''}</span>
        ))}
      </p>
      <p><strong>Director:</strong> {director.nombre} ({director.nacionalidad})</p>
      <p style={{ 
        color: calificacion >= 7 ? 'green' : calificacion >= 5 ? 'orange' : 'red',
        fontWeight: 'bold'
      }}>
        <strong>Calificación:</strong> {calificacion}/10
      </p>
      <div style={{ fontStyle: 'italic', marginTop: '10px' }}>
        {children}
      </div>
      <button onClick={() => onCambiarCalificacion(titulo, 10)}>
        ⭐ Calificar con 10
      </button>
    </div>
  );
}
